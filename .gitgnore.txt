# Unreal Engine 5 核心忽略项（必加）
/Saved/          # 忽略本地配置、日志、缓存
/Intermediate/   # 忽略编译中间文件
/Binaries/       # 忽略编译后的可执行文件
/DerivedDataCache/ # 忽略UE的派生数据缓存

# 编辑器配置忽略
/Config/WindowsEditor/ # 忽略Windows编辑器的本地配置
*.ini            # 可选：忽略所有ini配置（如果有自定义配置要提交，可删掉这行）

# 日志文件
*.log            # 忽略所有日志文件

# Visual Studio 相关（如果用VS开发）
*.sln
*.suo
.vs/
*.vcxproj
*.vcxproj.filters
*.vcxproj.user

# 系统自动生成文件
.DS_Store        # Mac系统文件
Thumbs.db        # Windows缩略图缓存
